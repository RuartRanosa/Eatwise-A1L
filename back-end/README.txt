	npm init								// create package.json

	npm install mysql						// install modules for connecting mysql to nodeJS

	npm install express --save 				// install express

	npm install express body-parser --save	// install express body-parser

	npm install request						// install request modules for testing http requests

	npm install cors

	sudo npm install -g create-react-app	// install react

// in index.js, change the parameters of "connection" depending on the configuration of your mariadb
// "node index.js" = start server
// "node request.js" = test created http requests


	create-react-app react-basics			// "react-basics" folder contains react modules
